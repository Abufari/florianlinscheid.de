import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from mpl_toolkits.mplot3d import axes3d

df1.head()

df3.head()

df2.head()

df4 = pd.read_csv('locations_4_manual.csv', names=['y', 'z'])

df2.head()
